//********** Key Data Structures **********\\
Board:
    squares[8][8] -> Piece or Empty
    
Piece:
    type: N, B, R, Q, K, P
    color: White, Black
    position: (row, col)

Move:
    piece: Piece
    from: (row, col)
    to: (row, col)

Path:
    squares: list of (row, col)  // ordered sequence of squares traversed


//********** Path Generation **********\\
The goal of path generation is to find the desired path of the piece, without taking blocking pieces into account.

function generatePath(move):
    piece = move.piece
    from = move.from
    to = move.to
    
    switch piece.type:
        case N:
            return generateKnightPath(from, to)
        case B:
            return generateDiagonalPath(from, to)
        case R:
            return generateOrthogonalPath(from, to)
        case Q:
            return generateQueenPath(from, to)
        case K:
            return generateKingPath(from, to)
        case P:
            return generatePawnPath(from, to)


function generateKnightPath(from, to):
    rowDiff = to.row - from.row
    colDiff = to.col - from.col
    
    // Build both possible L-paths
    pathVerticalFirst = []
    pathHorizontalFirst = []
    
    // Vertical first: move along rows, then columns
    current = from
    rowStep = sign(rowDiff)
    colStep = sign(colDiff)
    
    // Move vertically
    for i in 1 to abs(rowDiff):
        current = (current.row + rowStep, current.col)
        pathVerticalFirst.append(current)
    
    // Move horizontally
    for i in 1 to abs(colDiff):
        current = (current.row, current.col + colStep)
        pathVerticalFirst.append(current)
    
    // Horizontal first: move along columns, then rows
    current = from
    
    // Move horizontally
    for i in 1 to abs(colDiff):
        current = (current.row, current.col + colStep)
        pathHorizontalFirst.append(current)
    
    // Move vertically
    for i in 1 to abs(rowDiff):
        current = (current.row + rowStep, current.col)
        pathHorizontalFirst.append(current)
    
    return [pathVerticalFirst, pathHorizontalFirst]


function generateDiagonalPath(from, to):
    path = []
    rowStep = sign(to.row - from.row)
    colStep = sign(to.col - from.col)
    current = from
    
    while current != to:
        current = (current.row + rowStep, current.col + colStep)
        path.append(current)
    
    return [path]


function generateOrthogonalPath(from, to):
    path = []
    rowStep = sign(to.row - from.row)
    colStep = sign(to.col - from.col)
    current = from
    
    while current != to:
        current = (current.row + rowStep, current.col + colStep)
        path.append(current)
    
    return [path]


function generateQueenPath(from, to):
    // Determine if diagonal or orthogonal
    rowDiff = abs(to.row - from.row)
    colDiff = abs(to.col - from.col)
    
    if rowDiff == colDiff:
        return generateDiagonalPath(from, to)
    else:
        return generateOrthogonalPath(from, to)


function generateKingPath(from, to):
    // King moves one square, so path is just the destination
    return [[to]]


function generatePawnPath(from, to):
    // Similar to king, usually just one or two squares forward
    path = []
    rowStep = sign(to.row - from.row)
    current = from
    
    while current != to:
        current = (current.row + rowStep, current.col)
        path.append(current)
    
    return [path]

//********** Blocker Detection **********\\
function findBlockers(board, path):
  blockers = []

  //check all squares except for the last one (destination)
  //destination may have a capture, but we handle that separately
  for i in 0 to len(path) - 2:
    square = path[i]
    if board.squares[square.row][square.col] != Empty:
      blockers.append(square)
  
  return blockers

function selectBestPath(board, paths):
  bestPath = null
  fewestBlockers = infinity

  for path in paths:
    blockers = findBlockers(board, path)
    if len(blockers) < fewestBlockers:
      fewestBlockers = len(blockers)
      bestPath = path

  return bestPath

//********** Parking Spot Selection **********\\
function findParkingSpot(board, blockerPos, excludeSquares):
  //find nearest empty square that isn't in the primary path
  //use bfs to search outward from blockers position
  visited = set()
  queue = [blockerPos]

  while queue is not empty:
    current = queue.pop_front()

    if current in visited:
      continue
    visited.add(current)

    //check if valid parking spot
    if current != blockerPos:
      if board.squares[current.row][current.col] == Empty:
        if current not in excludeSquares:
          return current

    //add neighbors (all 8 directions)
    for neighbor in getNeighbors(current):
      if isValidSquare(neighbor) and neighbor not in visited:
        queue.append(neighbor)

  return null // no parking spot found

function getNeighbors(pos):
  neighbors = []
  for rowDelta in  [-1, 0, 1]:
    for colDelta in [-1, 0, 1]:
      if rowDelta == 0 and colDelta == 0:
        continue
      neighbors.append((pos.row + rowDelta, pos.col + colDelta))

  return neighbors

//********** Blocker Path Planning **********\\
function planBlockerMove(board, from, to):
    // Blockers can move any efficient route
    // Prefer diagonal movement when possible, use A* or BFS
    
    return findClearPath(board, from, to)


function findClearPath(board, from, to):
    // A* pathfinding avoiding occupied squares
    
    openSet = PriorityQueue()
    openSet.add(from, priority = heuristic(from, to))
    
    cameFrom = {}
    gScore = {from: 0}
    
    while openSet is not empty:
        current = openSet.pop()
        
        if current == to:
            return reconstructPath(cameFrom, current)
        
        for neighbor in getNeighbors(current):
            if not isValidSquare(neighbor):
                continue
            if board.squares[neighbor.row][neighbor.col] != Empty and neighbor != to:
                continue
            
            tentativeG = gScore[current] + 1
            
            if tentativeG < gScore.get(neighbor, infinity):
                cameFrom[neighbor] = current
                gScore[neighbor] = tentativeG
                openSet.add(neighbor, priority = tentativeG + heuristic(neighbor, to))
    
    return null  // No path found


function heuristic(from, to):
    // Chebyshev distance (allows diagonal movement)
    return max(abs(to.row - from.row), abs(to.col - from.col))


//********** Main Execution Planner **********\\
function planMove(board, move):
    // Generate possible paths for the primary piece
    paths = generatePath(move)
    
    // Select the path with fewest blockers
    primaryPath = selectBestPath(board, paths)
    
    // Find all blockers on the chosen path
    blockerSquares = findBlockers(board, primaryPath)
    
    // Plan relocation for each blocker
    relocationPlan = []
    excludeSquares = set(primaryPath)
    
    for blockerPos in blockerSquares:
        parkingSpot = findParkingSpot(board, blockerPos, excludeSquares)
        
        if parkingSpot == null:
            return Error("Cannot find parking spot for blocker")
        
        blockerPath = planBlockerMove(board, blockerPos, parkingSpot)
        
        if blockerPath == null:
            return Error("Cannot find clear path for blocker")
        
        relocationPlan.append({
            piece: board.squares[blockerPos.row][blockerPos.col],
            from: blockerPos,
            to: parkingSpot,
            path: blockerPath
        })
        
        // Mark parking spot as occupied for subsequent planning
        excludeSquares.add(parkingSpot)
    
    // Build final execution sequence
    return {
        relocations: relocationPlan,
        primaryMove: {
            piece: move.piece,
            path: primaryPath
        },
        restorations: reverse(relocationPlan)  // Return blockers in reverse order
    }

//********** Execution **********\\
function executeMovePlan(gantry, plan):
    // Move blockers out of the way
    for relocation in plan.relocations:
        gantry.movePiece(relocation.from, relocation.to, relocation.path)
    
    // Execute primary move
    gantry.movePiece(plan.primaryMove.from, plan.primaryMove.to, plan.primaryMove.path)
    
    // Restore blockers to original positions
    for restoration in plan.restorations:
        returnPath = planBlockerMove(board, restoration.to, restoration.from)
        gantry.movePiece(restoration.to, restoration.from, returnPath)
